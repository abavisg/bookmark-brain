```typescript
  updatedAt: number    // epoch ms — included for v2 sync compatibility

}

export interface PageContent {
  id: number
  bookmarkId: number   // FK to bookmarks.id
  rawText: string
  extractedAt: number  // epoch ms
}

export interface ProcessingJob {
  id: number
  bookmarkId: number   // FK to bookmarks.id
  status: JobStatus
  attempts: number     // starts at 0; incremented on each failed attempt
  nextRetryAt: number  // epoch ms — set to Date.now() for immediate; future for backoff
  lastError?: string   // stored on failure — used by requeueFailedApiKeyJobs to detect missing-key errors
  createdAt: number    // epoch ms
}
```

**src/shared/db/quota.ts**
`checkAndUpdateQuotaWarning(): Promise<void>` — reads `navigator.storage.estimate()`, computes `usage / quota`, calls `chrome.storage.local.set({ storageWarning: ratio >= 0.70 })`. Guard with `if (!navigator.storage?.estimate) return` for non-supporting contexts.

**src/shared/db/deviceId.ts**
`getOrCreateDeviceId(): Promise<string>` — reads `chrome.storage.local.get('deviceId')`. If present, returns it. If absent, generates `crypto.randomUUID()`, writes to `chrome.storage.local.set({ deviceId })`, and returns the new value. This ensures a stable ID across service worker restarts.
